\begin{quote}
Return an array free of duplicate values. Fastest E\+S5 implementation. \end{quote}


\subsection*{Install with \href{npmjs.org}{\tt npm}}


\begin{DoxyCode}
npm i array-unique --save
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
var unique = require('array-unique');

unique(['a', 'b', 'c', 'c']);
//=> ['a', 'b', 'c']
\end{DoxyCode}
