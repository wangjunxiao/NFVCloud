\begin{quote}
Convert a value to an array \end{quote}


\subsection*{Install}


\begin{DoxyCode}
$ npm install --save arrify
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
const arrify = require('arrify');

arrify('unicorn');
//=> ['unicorn']

arrify(['unicorn']);
//=> ['unicorn']

arrify(null);
//=> []

arrify(undefined);
//=> []
\end{DoxyCode}


{\itshape Supplying {\ttfamily null} or {\ttfamily undefined} results in an empty array.}
