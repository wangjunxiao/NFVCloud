\subsection*{Install}

With \href{https://npmjs.org/package/alphanum-sort}{\tt npm} do\+:


\begin{DoxyCode}
npm i alphanum-sort -S
\end{DoxyCode}


\subsection*{Example}


\begin{DoxyCode}
var sort = require('alphanum-sort');

var result = sort(['item20', 'item19', 'item1', 'item10', 'item2']);
// ['item1', 'item2', 'item10', 'item19', 'item20']
\end{DoxyCode}
