\begin{quote}
Check if a path is in the \href{http://en.wikipedia.org/wiki/Working_directory}{\tt current working directory} \end{quote}


\subsection*{Install}


\begin{DoxyCode}
$ npm install --save is-path-in-cwd
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
var isPathInCwd = require('is-path-in-cwd');

isPathInCwd('unicorn');
//=> true

isPathInCwd('../rainbow');
//=> false
\end{DoxyCode}
