\begin{quote}
Create an array of unique values, in order, from the input arrays \end{quote}


\subsection*{Install}


\begin{DoxyCode}
$ npm install --save array-union
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
const arrayUnion = require('array-union');

arrayUnion([1, 1, 2, 3], [2, 3]);
//=> [1, 2, 3]

arrayUnion(['foo', 'foo', 'bar'], ['foo']);
//=> ['foo', 'bar']
\end{DoxyCode}
