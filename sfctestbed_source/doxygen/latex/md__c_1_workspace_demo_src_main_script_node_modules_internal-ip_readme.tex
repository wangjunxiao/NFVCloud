\begin{quote}
Get your internal I\+Pv4 or I\+Pv6 address \end{quote}


\subsection*{C\+LI}


\begin{DoxyCode}
$ npm install --global internal-ip
\end{DoxyCode}



\begin{DoxyCode}
$ internal-ip --help

  Usage
    $ internal-ip

  Options
    -4, --ipv4  Return the IPv4 address (default)
    -6, --ipv6  Return the IPv6 address

  Example
    $ internal-ip
    192.168.0.123
    $ internal-ip -6
    fe80::200:f8ff:fe21:67cf
\end{DoxyCode}
