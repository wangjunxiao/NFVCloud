\begin{quote}
Require a module bypassing the \href{https://nodejs.org/api/modules.html#modules_caching}{\tt cache} \end{quote}


Useful for testing purposes when you need to freshly require a module.

\subsection*{Install}


\begin{DoxyCode}
$ npm install --save require-uncached
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
// foo.js
let i = 0;
module.exports = () => ++i;
\end{DoxyCode}



\begin{DoxyCode}
const requireUncached = require('require-uncached');

require('./foo')();
//=> 1

require('./foo')();
//=> 2

requireUncached('./foo')();
//=> 1

requireUncached('./foo')();
//=> 1
\end{DoxyCode}
