Parsing the Java\+Script\textquotesingle{}s Reg\+Exp in Java\+Script.

\subsection*{Installation}


\begin{DoxyCode}

\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}

var parseTree = parse('^a'); // /^a/
console.log(parseTree);
\end{DoxyCode}
