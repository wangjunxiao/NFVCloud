Merges the properties from a source object into a destination object.

\subsection*{Install}

\begin{DoxyVerb}$ npm install utils-merge
\end{DoxyVerb}


\subsection*{Usage}


\begin{DoxyCode}
var a = \{ foo: 'bar' \}
  , b = \{ bar: 'baz' \};

merge(a, b);
// => \{ foo: 'bar', bar: 'baz' \}
\end{DoxyCode}
