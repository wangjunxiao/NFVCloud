\begin{quote}
Check if a path is inside another path \end{quote}


\subsection*{Install}


\begin{DoxyCode}
$ npm install --save is-path-inside
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
var isPathInside = require('is-path-inside');

isPathInside('a/b', 'a/b/c');
//=> true

isPathInside('x/y', 'a/b/c');
//=> false

isPathInside('a/b/c', 'a/b/c');
//=> false
\end{DoxyCode}
