\begin{quote}
Regular expression for matching file names, with or without extension. \end{quote}


\subsection*{Install}

Install with \href{https://www.npmjs.com/}{\tt npm}\+:


\begin{DoxyCode}
$ npm install --save filename-regex
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
var regex = require('filename-regex');

'a/b/c/d.min.js'.match(regex());
//=> match[0] = 'd.min.js'

'a/b/c/.dotfile'.match(regex());
//=> match[0] = '.dotfile'
\end{DoxyCode}
